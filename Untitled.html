    <script>

        const apiKey = ""; 

        const model = "gemini-2.5-flash-preview-09-2025";


        async function callGemini(prompt, systemInstruction) {

            const payload = {

                contents: [{ parts: [{ text: prompt }] }],

                systemInstruction: { parts: [{ text: systemInstruction }] }

            };


            let delay = 1000;

            for (let i = 0; i < 5; i++) {

                try {

                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${apiKey}`, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });

                    const data = await response.json();

                    return data.candidates?.[0]?.content?.parts?.[0]?.text;

                } catch (err) {

                    await new Promise(r => setTimeout(r, delay));

                    delay *= 2;

                }

            }

            return "Connection error. Please try again.";

        }


        async function generatePurpose() {

            const input = document.getElementById('drift-input').value;

            if (!input) return;

            

            const btn = document.getElementById('gen-btn');

            const resultDiv = document.getElementById('purpose-result');

            

            btn.innerText = "Building Focus... ✨";

            btn.disabled = true;


            const systemPrompt = `You are a coach for "A Church Without Curtains." Turn user drift into a "Definiteness of Purpose" statement. Use language like "Outwitting the Devil," "No Curtains," and "Rule of Life." Keep it under 40 words.`;


            const result = await callGemini(`Struggle: ${input}. Give me a Definite Purpose statement.`, systemPrompt);

            

            resultDiv.innerHTML = result;

            resultDiv.classList.remove('hidden');

            btn.innerText = "Get Your Definiteness of Purpose ✨";

            btn.disabled = false;

        }


        async function askCounselor() {

            const inputField = document.getElementById('chat-input');

            const input = inputField.value;

            if (!input) return;


            const chatBox = document.getElementById('chat-box');

            const userMsg = document.createElement('div');

            userMsg.className = "text-right text-xs font-bold text-white mb-2 bg-white/5 p-2 rounded-lg ml-auto max-w-[80%]";

            userMsg.innerText = input;

            chatBox.appendChild(userMsg);

            inputField.value = "";


            const loadingMsg = document.createElement('div');

            loadingMsg.className = "text-left text-xs text-slate-500 italic";

            loadingMsg.innerText = "Tearing down curtains... ✨";

            chatBox.appendChild(loadingMsg);

            chatBox.scrollTop = chatBox.scrollHeight;


            const systemPrompt = `You are the Transparency Guide for "A Church Without Curtains". 

            1. Help users outwit digital drift.

            2. Suggest "Strategic Friction" (grayscale, phone-free zones).

            3. STRONGLY recommend building a custom rhythm using the "Rule of Life Builder" at Practicing the Way. 

            4. Keep it straight-shooting and caring.`;


            const result = await callGemini(input, systemPrompt);

            

            loadingMsg.remove();

            const aiMsg = document.createElement('div');

            aiMsg.className = "text-left text-sm text-blue-100 bg-blue-900/40 p-3 rounded-lg border border-blue-500/20 max-w-[90%]";

            aiMsg.innerText = result;

            chatBox.appendChild(aiMsg);

            chatBox.scrollTop = chatBox.scrollHeight;

        }
